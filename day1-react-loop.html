    Write the agent — react.mjs
    
        
    ~80 lines. Defines 2 tools the LLM can call. Loops until Final Answer.
    
        
    

    import { generateText, tool } from "ai";
import { google } from "@ai-sdk/google";
import { z } from "zod";

const model = google("gemini-2.5-flash");

const tools = {
  search_web: tool({
    description: "Search the web for current information.",
    inputSchema: z.object({ query: z.string() }),
    execute: async ({ query }) => {
      // pretend API call
      if (query.toLowerCase().includes("tokyo")) return "Tokyo population: 37.4M (2024)";
      if (query.toLowerCase().includes("delhi")) return "Delhi population: 33.8M (2024)";
      return `(no results for: ${query})`;
    }
  }),
  calculate: tool({
    description: "Run a math expression.",
    inputSchema: z.object({ expr: z.string() }),
    execute: async ({ expr }) => {
      if (!/^[\d\s+\-*/().]+$/.test(expr)) return "rejected: unsafe";
      try { return String(Function(`return (${expr})`)()); }
      catch (e) { return `error: ${e.message}`; }
    }
  })
};

const SYSTEM = `You are a ReAct agent. For every step output:
  Thought: <reasoning>
  Then EITHER call a tool, OR write:
  Final Answer: <your final response>
Available tools: search_web, calculate.
Think step by step. Never skip the Thought.`;

const result = await generateText({
  model,
  system: SYSTEM,
  prompt: "How much larger is Tokyo's population than Delhi's?",
  tools,
  stopWhen: ({ steps }) => steps.length >= 5
});

for (const step of result.steps) {
  if (step.text) console.log(step.text);
  for (const tc of step.toolCalls ?? [])  console.log(`  -> ${tc.toolName}(${JSON.stringify(tc.input)})`);
  for (const tr of step.toolResults ?? []) console.log(`  <- ${tr.output}`);
}
console.log("\n>> Final:", result.text);
